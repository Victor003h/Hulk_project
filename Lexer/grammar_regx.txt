Regex       -> Union
Union       -> Concat Union'
Union'      -> '|' Concat Union' | ε
Concat      -> Repeat Concat'
Concat'     -> Repeat Concat' | ε
Repeat      -> Atom Repeat'
Repeat'     -> '*' Repeat' | ε
Atom        -> '(' Regex ')' | SYMBOL


Regex     → Term Regex'
Regex'    → '|' Term Regex' | ε
Term      → Factor Term'
Term'     → Factor Term' | ε
Factor    → Base Factor'
Factor'   → '*' Factor' | ε
Base      → '(' Regex ')' | symbol
